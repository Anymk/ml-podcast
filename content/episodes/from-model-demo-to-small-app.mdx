## 直接运行

这个例子会判断一句评论更像 `positive` 还是 `negative`。

<CodePlayground
  title="Naive Bayes Demo"
  description="这段代码先统计正面和负面句子里的单词，再对新句子做分类。你可以直接修改 testText。"
  code={`const trainingData = [
  { text: "good course clear lesson", label: "positive" },
  { text: "helpful teacher nice examples", label: "positive" },
  { text: "great tutorial easy to follow", label: "positive" },
  { text: "bad course confusing lesson", label: "negative" },
  { text: "boring teacher unclear examples", label: "negative" },
  { text: "hard tutorial not helpful", label: "negative" }
];

function tokenize(text) {
  return text.toLowerCase().split(" ");
}

function trainNaiveBayes(data) {
  const wordCounts = { positive: {}, negative: {} };
  const docCounts = { positive: 0, negative: 0 };
  const totalWords = { positive: 0, negative: 0 };
  const vocabulary = new Set();

  for (const item of data) {
    docCounts[item.label] += 1;
    const tokens = tokenize(item.text);

    for (const token of tokens) {
      vocabulary.add(token);
      wordCounts[item.label][token] = (wordCounts[item.label][token] || 0) + 1;
      totalWords[item.label] += 1;
    }
  }

  return { wordCounts, docCounts, totalWords, vocabularySize: vocabulary.size, totalDocs: data.length };
}

function scoreLabel(model, text, label) {
  const tokens = tokenize(text);
  let score = Math.log(model.docCounts[label] / model.totalDocs);

  for (const token of tokens) {
    const count = model.wordCounts[label][token] || 0;
    const probability = (count + 1) / (model.totalWords[label] + model.vocabularySize);
    score += Math.log(probability);
  }

  return score;
}

function predictSentiment(model, text) {
  const positiveScore = scoreLabel(model, text, "positive");
  const negativeScore = scoreLabel(model, text, "negative");

  return positiveScore > negativeScore ? "positive" : "negative";
}

const model = trainNaiveBayes(trainingData);
const testText = "clear helpful lesson";
const prediction = predictSentiment(model, testText);

console.log("Text:", testText);
console.log("Prediction:", prediction);`}
